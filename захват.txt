#include <Servo.h>

Servo servo1;
Servo servo2;

void setup() {
  servo1.attach(9); // Подключаем первый сервомотор к пину 9
  servo2.attach(10); // Подключаем второй сервомотор к пину 10
}

void loop() {
  // Допустим, что камера ESP32 определила объект перед собой

  // Захватываем объект
  servo1.write(90); // Поворачиваем первый сервомотор
  delay(1000); // Ждем 1 секунду
  servo2.write(90); // Поворачиваем второй сервомотор
  delay(1000); // Ждем 1 секунду

  // Поднимаем объект на 10 см
  for (int i = 0; i < 10; i++) {
    servo2.write(180);
    delay(100);
  }

  // Держим объект 5 секунд
  delay(5000);

  // Ставим объект обратно на место
  for (int i = 0; i < 10; i++) {
    servo2.write(90);
    delay(100);
  }

  // Возвращаемся в исходное состояние
  servo2.write(0);
  delay(1000);
  servo1.write(0);
  delay(1000);
}
